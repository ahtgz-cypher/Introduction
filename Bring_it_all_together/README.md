def bytes2matrix(text):

    return [list(text[i::4]) for i in range(4)]

def matrix2bytes(matrix):

    return bytes(b"".join(bytes[a] for row in matrix for a in row))

def add_round_key(s, k):

    return [[s[i][j]^k[i][j] for j in range(4)] for i in range(4)]

def inv_shift_row(s):

    s[1][1], s[2][1], s[3][1], s[0][1] = s[0][1], s[1][1], s[2][1], s[3][1]
    s[2][2], s[3][2], s[0][2], s[1][2] = s[0][2], s[1][2], s[2][2], s[3][2]
    s[3][3], s[0][3], s[1][3], s[2][3] = s[0][3], s[1][3], s[2][3], s[3][3]

xtime = lambda a: (((a << 1) ^ 0x1B) & 0xFF) if (a & 0x80) else (a << 1)

def mix_single_column(a):

    t = a[0] ^ a[1] ^ a[2] ^ a[3]
    u = a[0]
    a[0] ^= t ^ xtime(a[0] ^ a[1])
    a[1] ^= t ^ xtime(a[1] ^ a[2])
    a[2] ^= t ^ xtime(a[2] ^ a[3])
    a[3] ^= t ^ xtime(a[3] ^ u)

def mix_column(s):

    for i in range(4):
        mix_single_column(s[i])

def inv_mix_column(s):

    for i in range(4):
        u = xtime(xtime(s[i][0] ^ s[i][2]))
        v = xtime(xtime(s[i][1] ^ s[i][3]))
        s[i][0] ^= u
        s[i][1] ^= v
        s[i][2] ^= u
        s[i][3] ^= v
    mix_column(s)

def sub_bytes(s, sbox = s_box):

    return [[int(sbox[a]) for a in row] for row in s]

        inv_shift_row(state)
        state = sub_bytes(state, sbox = inv_s_box)

    state = add_round_key(state, round_keys.pop())
    # Convert state matrix to plaintext
    plaintext = matrix2bytes(state)

    return plaintext
